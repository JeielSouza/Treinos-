double euler(int x)
{
    int nur = 0; 
    float result = 0;
    int i; 
    int fators = 0;
    int somas = 0; 

    for (i = 0; i < x + 1; i++)
    {
        if ( i == 0 )
        {
            fators = 1; 
        }
        else 
        {
            fators = fators * i; 
            somas = somas + fators; 
        }

        nur = nur + 1; 
    }
    
    result = nur / somas; 

    return result; 
    
}
